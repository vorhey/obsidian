```go
  /* arithmetic */
  + // addition
  - // substraction
  / // division
  * // multiplication
  % // modulus
  ++ // increment
  -- // decrement

  /* comparison */
  == // equal
  != // not equal
  < // less than
  > // greater than
  <= // less than or equal
  >= // greater than or equal

  /* logical */
  && // and
  || // or
  ! // not

  /* bitwise */
  & // and
  | // or
  ^ // xor
  << // left shift
  >> // right shift
  &^ // and not

  /* assignment */
  = // assignment
  += // addition assignment
  -= // substraction assignment
  *= // multiplication assignment
  /= // division assignment
  %= // modulus assignment
  <<= // left shift assignment
  >>= // right shift assignment
  &= // and assignment
  ^= // xor assignment
  |= // or assignment

  /* address and pointer */
  & // address
  * // pointer

  /* channel */
  <- // receive

```
